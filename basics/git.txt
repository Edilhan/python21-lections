git init - создаем git репозиторий из текущ папки
git remote add название ssh-ссылка - сохраняем ssh-ссылку в переменную
git remote -v - посмотр все сохр ssh-ссылки
git add . - прост пишем эту комманду кажд раз перед commit
git commit -m 'message' - созд нов версию
git push orogin master - закидыв версию в github
git clone ssh-ссылка - копируем репозиторий с гитхаба
git log - показ все коммиты
git branch - показ все ветки и выделяет звездочкой текущ ветку
git branch название - создает ветку
git checkout ветка - перейти на ветку
git checkout -b - создать и перейти
git branch -D - удалить ветку



!!!!!!WARNING!!!!!!
git reset HEAD~ - отменяет последний коммит, как коммит но наоборот
git restore . - похоже на add, но наоборот
